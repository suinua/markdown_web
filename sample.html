コードブロック

class Article {
  final String uuid;
  final String title;
  final String body;
  final List<String> tags;

  const Article({
    required this.uuid,
    required this.title,
    required this.body,
    required this.tags,
  });

  @override
  bool operator ==(Object other) {
    if (other is Article) {
      return other.uuid == uuid;
    }

    return false;
  }
}
